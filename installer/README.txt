المتطلبات:
- Windows 10 أو أحدث
- Node.js (يُفضل تثبيته من: https://nodejs.org)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

التنصيب:
1. اضغط مرتين على "setup.bat"
2. أدخل رقم الهوست (عنوان IP للجهاز)
   - مثال: 192.168.1.100
   - أو أدخل 0.0.0.0 للسماح بالوصول من جميع الأجهزة
3. أدخل رقم المنفذ (اتركه فارغ للمنفذ 4357)
4. اضغط Y للمتابعة
5. انتظر حتى يكمل التنصيب

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

التشغيل:
- اضغط على اختصار "نظام القيود اليومية" على سطح المكتب
- أو اضغط مرتين على "تشغيل.bat" في مجلد التنصيب

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

بيانات الدخول الافتراضية:
  اسم المستخدم: admin
  كلمة المرور:   admin123

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الوصول من أجهزة أخرى:
1. تأكد أن الجهازين على نفس الشبكة
2. افتح المتصفح على الجهاز الآخر
3. اكتب: http://رقم-الهوست:المنفذ
   مثال: http://192.168.1.100:4357
